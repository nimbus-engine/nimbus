Using Plugins

<!-- Math operations -->
<Plugin Name="MathPlugin" Method="sqrt" Params="16" ToState="result"/>

<!-- String manipulation -->
<Plugin Name="StringPlugin" Method="upper" Params="{text}" ToState="upperText"/>

<!-- Generate hash -->
<Plugin Name="CryptoPlugin" Method="md5" Params="{password}" ToState="hash"/>
